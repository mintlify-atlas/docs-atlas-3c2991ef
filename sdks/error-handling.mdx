All Rexec SDKs provide structured error handling to help you build robust applications. This guide covers error types, handling patterns, and best practices for each SDK.

## Error Types

Rexec SDKs expose different error types depending on the failure mode:

<AccordionGroup>
  <Accordion title="API Errors (4xx/5xx)">
    HTTP errors returned by the Rexec API, including:
    - `400 Bad Request` - Invalid request parameters
    - `401 Unauthorized` - Invalid or expired token
    - `403 Forbidden` - Insufficient permissions
    - `404 Not Found` - Resource doesn't exist
    - `429 Too Many Requests` - Rate limit exceeded
    - `500 Internal Server Error` - Server error
  </Accordion>
  
  <Accordion title="Network Errors">
    Connection failures, timeouts, or DNS resolution errors when reaching the Rexec instance.
  </Accordion>
  
  <Accordion title="WebSocket Errors">
    Terminal connection failures, including disconnections and protocol errors.
  </Accordion>
  
  <Accordion title="Validation Errors">
    Client-side validation failures before making API requests.
  </Accordion>
</AccordionGroup>

## SDK-Specific Error Handling

### Go SDK

The Go SDK returns structured errors that implement the standard `error` interface.

```go
package main

import (
    "context"
    "fmt"
    rexec "github.com/PipeOpsHQ/rexec-go"
)

func main() {
    client := rexec.NewClient(baseURL, token)
    ctx := context.Background()
    
    container, err := client.Containers.Get(ctx, "invalid-id")
    if err != nil {
        // Check if it's an API error
        if apiErr, ok := err.(*rexec.APIError); ok {
            fmt.Printf("API Error %d: %s\n", apiErr.StatusCode, apiErr.Message)
            
            // Handle specific status codes
            switch apiErr.StatusCode {
            case 404:
                fmt.Println("Container not found")
            case 401:
                fmt.Println("Invalid token - please re-authenticate")
            case 429:
                fmt.Println("Rate limit exceeded - retry later")
            default:
                fmt.Printf("Unexpected error: %v\n", apiErr)
            }
        } else {
            // Network or other error
            fmt.Printf("Request failed: %v\n", err)
        }
        return
    }
}
```

<Note>
The Go SDK's `APIError` type includes both `StatusCode` and `Message` fields for detailed error inspection.
</Note>

### JavaScript/TypeScript SDK

The TypeScript SDK provides a custom `RexecError` class for API errors.

```typescript
import { RexecClient, RexecError } from '@pipeopshq/rexec';

const client = new RexecClient({ baseURL, token });

try {
  const container = await client.containers.get('invalid-id');
} catch (error) {
  if (error instanceof RexecError) {
    console.error(`API Error ${error.statusCode}: ${error.message}`);
    
    // Handle specific errors
    switch (error.statusCode) {
      case 404:
        console.log('Container not found');
        break;
      case 401:
        console.log('Authentication failed - refresh token');
        break;
      case 429:
        console.log('Rate limited - implement exponential backoff');
        break;
      default:
        console.error('Unexpected API error:', error);
    }
  } else {
    // Network error
    console.error('Network error:', error);
  }
}
```

#### Retry with Exponential Backoff

```typescript
async function createContainerWithRetry(
  client: RexecClient,
  options: CreateContainerRequest,
  maxRetries = 3
): Promise<Container> {
  let lastError: Error;
  
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await client.containers.create(options);
    } catch (error) {
      lastError = error;
      
      if (error instanceof RexecError) {
        // Don't retry client errors (4xx)
        if (error.statusCode >= 400 && error.statusCode < 500) {
          throw error;
        }
        
        // Retry server errors (5xx) with exponential backoff
        if (error.statusCode >= 500) {
          const delay = Math.pow(2, i) * 1000;
          console.log(`Retry ${i + 1}/${maxRetries} after ${delay}ms`);
          await new Promise(resolve => setTimeout(resolve, delay));
          continue;
        }
      }
      
      throw error;
    }
  }
  
  throw lastError;
}
```

### Python SDK

The Python SDK provides structured exception classes.

```python
from rexec import RexecClient, RexecAPIError, RexecConnectionError
import asyncio

async def main():
    async with RexecClient(base_url, token) as client:
        try:
            container = await client.containers.get("invalid-id")
        except RexecAPIError as e:
            print(f"API Error {e.status_code}: {e.message}")
            
            # Handle specific status codes
            if e.status_code == 404:
                print("Container not found")
            elif e.status_code == 401:
                print("Authentication failed")
            elif e.status_code == 429:
                print("Rate limit exceeded")
                # Implement retry logic
                await asyncio.sleep(60)
        except RexecConnectionError as e:
            print(f"Connection error: {e}")
            # Check network connectivity
        except Exception as e:
            print(f"Unexpected error: {e}")

asyncio.run(main())
```

#### Context Manager for Cleanup

```python
from rexec import RexecClient
import asyncio

async def safe_container_operation():
    async with RexecClient(base_url, token) as client:
        container = None
        try:
            # Create container
            container = await client.containers.create(
                image="ubuntu:24.04",
                name="test-container"
            )
            
            # Perform operations
            await client.containers.start(container.id)
            result = await client.containers.exec(container.id, "echo hello")
            print(result.stdout)
            
        except Exception as e:
            print(f"Error during operation: {e}")
        finally:
            # Always cleanup
            if container:
                try:
                    await client.containers.delete(container.id)
                except Exception as e:
                    print(f"Cleanup failed: {e}")
```

### Rust SDK

The Rust SDK uses the `Result` type with custom error enums.

```rust
use rexec::{RexecClient, Error};

#[tokio::main]
async fn main() {
    let client = RexecClient::new(base_url, token);
    
    match client.containers().get("invalid-id").await {
        Ok(container) => println!("Found: {}", container.name),
        Err(Error::Api { status_code, message }) => {
            eprintln!("API Error {}: {}", status_code, message);
            
            match status_code {
                404 => eprintln!("Container not found"),
                401 => eprintln!("Authentication failed"),
                429 => eprintln!("Rate limited"),
                _ => eprintln!("Unexpected error"),
            }
        }
        Err(Error::Connection(msg)) => {
            eprintln!("Connection error: {}", msg);
        }
        Err(e) => eprintln!("Error: {}", e),
    }
}
```

#### Using the `?` Operator

```rust
use rexec::{RexecClient, Error, CreateContainerRequest};

async fn create_and_start(client: &RexecClient) -> Result<String, Error> {
    // Errors propagate automatically with ?
    let container = client.containers()
        .create(CreateContainerRequest::new("ubuntu:24.04"))
        .await?;
    
    client.containers().start(&container.id).await?;
    
    Ok(container.id)
}

#[tokio::main]
async fn main() {
    let client = RexecClient::new(base_url, token);
    
    match create_and_start(&client).await {
        Ok(id) => println!("Container started: {}", id),
        Err(e) => eprintln!("Failed: {}", e),
    }
}
```

## Rate Limiting

Rexec enforces rate limits to ensure fair usage:

<Card title="Rate Limits" icon="gauge">
  - **Container creation**: 10 per minute
  - **API requests**: 100 per minute
  - **WebSocket connections**: 5 concurrent per user
</Card>

### Handling Rate Limits

When you exceed rate limits, the API returns a `429 Too Many Requests` status. Implement exponential backoff:

```typescript
async function withRateLimit<T>(
  fn: () => Promise<T>,
  maxRetries = 5
): Promise<T> {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn();
    } catch (error) {
      if (error instanceof RexecError && error.statusCode === 429) {
        const delay = Math.pow(2, i) * 1000; // Exponential backoff
        console.log(`Rate limited. Retrying in ${delay}ms...`);
        await new Promise(resolve => setTimeout(resolve, delay));
        continue;
      }
      throw error;
    }
  }
  throw new Error('Max retries exceeded');
}

// Usage
const container = await withRateLimit(() => 
  client.containers.create({ image: 'ubuntu:24.04' })
);
```

## Best Practices

<AccordionGroup>
  <Accordion title="Always Clean Up Resources">
    Use defer, finally, or context managers to ensure containers are deleted even if errors occur.
    
    ```go
    container, err := client.Containers.Create(ctx, req)
    if err != nil {
        return err
    }
    defer client.Containers.Delete(ctx, container.ID)
    ```
  </Accordion>
  
  <Accordion title="Implement Retry Logic">
    Retry transient errors (5xx, network issues) with exponential backoff. Don't retry client errors (4xx).
  </Accordion>
  
  <Accordion title="Handle Rate Limits Gracefully">
    Implement exponential backoff for 429 responses. Consider request queuing for high-volume applications.
  </Accordion>
  
  <Accordion title="Log Errors with Context">
    Include request IDs, container IDs, and timestamps in error logs for debugging.
    
    ```typescript
    console.error('Failed to create container', {
      image: 'ubuntu:24.04',
      error: error.message,
      statusCode: error.statusCode,
      timestamp: new Date().toISOString(),
    });
    ```
  </Accordion>
  
  <Accordion title="Use Timeouts">
    Set appropriate timeouts for operations to prevent hanging indefinitely.
    
    ```go
    ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
    defer cancel()
    ```
  </Accordion>
</AccordionGroup>
