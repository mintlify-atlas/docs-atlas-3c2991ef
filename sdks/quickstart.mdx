This guide walks you through setting up and using a Rexec SDK to programmatically interact with sandboxed environments.

## Prerequisites

<Steps>
  <Step title="A running Rexec instance">
    Self-hosted or cloud instance accessible via URL
  </Step>
  <Step title="An API token">
    Generate one from your Rexec dashboard (see below)
  </Step>
  <Step title="Your preferred language installed">
    Go, JavaScript, Python, Rust, Ruby, Java, .NET, or PHP
  </Step>
</Steps>

## Getting an API Token

Before using any SDK, you need to generate an API token:

<Steps>
  <Step title="Log in to your Rexec dashboard">
    Navigate to your Rexec instance in your web browser
  </Step>
  
  <Step title="Go to Settings → API Tokens">
    Find the API Tokens section in your account settings
  </Step>
  
  <Step title="Click Generate Token">
    Create a new API token with appropriate permissions
  </Step>
  
  <Step title="Copy and store the token securely">
    Save the token immediately - you won't be able to see it again
  </Step>
</Steps>

<Warning>
  **Security**: Never commit API tokens to version control. Use environment variables or secret management tools.
</Warning>

```bash
export REXEC_URL="https://your-instance.com"
export REXEC_TOKEN="your-api-token"
```

## Step-by-Step Tutorial

### Step 1: Create a Client

Initialize the SDK client with your instance URL and API token:

<Tabs>
  <Tab title="Go">
    ```go
    import rexec "github.com/PipeOpsHQ/rexec-go"

    client := rexec.NewClient(
        os.Getenv("REXEC_URL"),
        os.Getenv("REXEC_TOKEN"),
    )
    ```
  </Tab>
  
  <Tab title="JavaScript">
    ```typescript
    import { RexecClient } from '@pipeopshq/rexec';

    const client = new RexecClient({
      baseURL: process.env.REXEC_URL,
      token: process.env.REXEC_TOKEN,
    });
    ```
  </Tab>
  
  <Tab title="Python">
    ```python
    from rexec import RexecClient

    client = RexecClient(
        base_url=os.environ["REXEC_URL"],
        token=os.environ["REXEC_TOKEN"]
    )
    ```
  </Tab>
  
  <Tab title="Rust">
    ```rust

    let client = RexecClient::new(
        std::env::var("REXEC_URL")?,
        std::env::var("REXEC_TOKEN")?
    )?;
    ```
  </Tab>
  
  <Tab title="Ruby">
    ```ruby

    client = Rexec::Client.new(
      ENV['REXEC_URL'],
      ENV['REXEC_TOKEN']
    )
    ```
  </Tab>
  
  <Tab title="Java">

        System.getenv("REXEC_URL"),

  </Tab>
  
  <Tab title=".NET">

### Step 5: Work with Files

List, read, and write files in your container:

<CodeGroup>
```go Go - List Files
files, err := client.Files.List(ctx, container.ID, "/home")
if err != nil {
    log.Fatal(err)
}
for _, file := range files {
    fmt.Printf("%s - %d bytes\n", file.Name, file.Size)
}
```

```typescript JavaScript - Read File
const content = await client.files.read(
  container.id, 
  '/etc/hostname'
);
console.log(content);
```

```python Python - Write File
await client.files.write(
    container.id,
    "/app/script.sh",
    "#!/bin/bash\necho hello"
)
```

```rust Rust - Download File
let content = client.files()
    .download(&container.id, "/etc/passwd")
    .await?;
println!("{}", String::from_utf8_lossy(&content));
```
</CodeGroup>

### Step 6: Interactive Terminal

Connect to a real-time terminal session:

<Tabs>
  <Tab title="Go">
    ```go
    term, err := client.Terminal.Connect(ctx, container.ID)
    if err != nil {
        log.Fatal(err)
    }
    defer term.Close()

    term.OnData(func(data []byte) {
        fmt.Print(string(data))
    })

    term.Write([]byte("ls -la\n"))
    ```
  </Tab>
  
  <Tab title="JavaScript">
    ```typescript
    const terminal = await client.terminal.connect(container.id);

    terminal.onData((data) => process.stdout.write(data));
    terminal.write('ls -la\n');

    // Later: terminal.close();
    ```
  </Tab>
  
  <Tab title="Python">
    ```python
    async with client.terminal.connect(container.id) as term:
        await term.write(b"ls -la\n")
        async for data in term:
            print(data.decode(), end="")
    ```
  </Tab>
  
  <Tab title="Rust">
    ```rust
    let mut term = client.terminal()
        .connect(&container.id)
        .await?;

    term.write(b"ls -la\n").await?;

    while let Some(data) = term.read().await? {
        print!("{}", String::from_utf8_lossy(&data));
    }
    ```
  </Tab>
</Tabs>

## Complete Example

Here's a full working example:

<Tabs>
  <Tab title="Go">
    ```go
    package main

    import (
        "context"
        "fmt"
        "log"
        "os"
        rexec "github.com/PipeOpsHQ/rexec-go"
    )

    func main() {
        client := rexec.NewClient(
            os.Getenv("REXEC_URL"),
            os.Getenv("REXEC_TOKEN"),
        )

        ctx := context.Background()

        // Create container
        container, err := client.Containers.Create(ctx, &rexec.CreateContainerRequest{
            Image: "ubuntu:24.04",
            Name:  "my-sandbox",
        })
        if err != nil {
            log.Fatal(err)
        }
        defer client.Containers.Delete(ctx, container.ID)

        // Start it
        if err := client.Containers.Start(ctx, container.ID); err != nil {
            log.Fatal(err)
        }

        // Run a command
        result, err := client.Containers.Exec(ctx, container.ID, []string{"echo", "Hello!"})
        if err != nil {
            log.Fatal(err)
        }
        fmt.Println(result.Stdout)
    }
    ```
  </Tab>
  
  <Tab title="JavaScript">
    ```typescript
    import { RexecClient } from '@pipeopshq/rexec';

    const client = new RexecClient({
      baseURL: process.env.REXEC_URL!,
      token: process.env.REXEC_TOKEN!,
    });

    async function main() {
      // Create container
      const container = await client.containers.create({
        image: 'ubuntu:24.04',
        name: 'my-sandbox',
      });

      try {
        // Start it
        await client.containers.start(container.id);

        // Run a command
        const result = await client.containers.exec(
          container.id, 
          'echo "Hello!"'
        );
        console.log(result.stdout);
      } finally {
        // Clean up
        await client.containers.delete(container.id);
      }
    }

    main().catch(console.error);
    ```
  </Tab>
  
  <Tab title="Python">
    ```python
    import asyncio
    import os
    from rexec import RexecClient

    async def main():
        async with RexecClient(
            base_url=os.environ["REXEC_URL"],
            token=os.environ["REXEC_TOKEN"]
        ) as client:
            # Create container
            container = await client.containers.create(
                image="ubuntu:24.04",
                name="my-sandbox"
            )

            try:
                # Start it
                await client.containers.start(container.id)

                # Run a command
                result = await client.containers.exec(
                    container.id, 
                    "echo 'Hello!'"
                )
                print(result.stdout)
            finally:
                # Clean up
                await client.containers.delete(container.id)

    asyncio.run(main())
    ```
  </Tab>
</Tabs>
